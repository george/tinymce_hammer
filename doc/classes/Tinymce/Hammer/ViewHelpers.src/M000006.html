  def tinymce_hammer_javascript_tags

    init = Tinymce::Hammer.init.collect{|key,value|
      "#{key} : #{value.to_json}"
    }.join(', ')

    setup = "init.setup = #{Tinymce::Hammer.setup};" if Tinymce::Hammer.setup
    init = "<script type=\"text/javascript\">\nTinymceHammer = {\ninit : function() {\nvar init = { \#{init} };\ninit.mode = 'specific_textareas';\ninit.editor_selector = 'tinymce';\ninit.plugins = '\#{Tinymce::Hammer.plugins.join(',')}';\ninit.language = '\#{Tinymce::Hammer.languages.first}';\n\#{setup}\ntinyMCE.init(init);\n},\ninitOnLoad : function () {\nvar onloadAlias = window.onload;\nif (typeof window.onload != 'function') {\nwindow.onload = TinymceHammer.init;\n} else {\nwindow.onload = function() {\nif (onloadAlias) {\nonloadAlias();\n}\nTinymceHammer.init();\n}\n}\n}\n};\nTinymceHammer.initOnLoad();\n</script>\n"
    [javascript_include_tag(tinymce_hammer_js_path), init].join("\n")
  end
